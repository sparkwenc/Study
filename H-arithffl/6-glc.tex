\sct{Geometric Langlands}

Ref:~[Gaitsgory-Raskin].


\spg{de Rham setting}
\begin{stp}
	Let $k$ be a field of characteristic $0$, $X$ a smooth and complete curve over $k$, $G$ a connected reductive group over $k$, and $\oH{G}$ its Langlands dual over $k$.
\end{stp}

de Rham geometric Langlands: the following functor is an equivalence.
\[
	\begin{tikzcd}
		\text{D-Mod}_{\frac{1}{2}}(\Bun_{G})\ar[r, "\bL_{G}"]\ar[d]\ar[dr, "\bL_{G, \text{coarse}}"] & \IndCoh_{\text{Nilp}}(\mathrm{LS}_{\oH{G}})\ar[d, "{\Psi_{\text{Nilp}, \{0\}}}"] \\
		\Vsp & \Qch(\text{LS}_{\oH{G}})\ar[l, "{\Gma(\mathrm{LS}_{\oH{G}},-)}"]
	\end{tikzcd}
\]

Step1, construct $\bL_{G, \text{coarse}}$

Step2, lift to $\bL_{G}$, (Theorem~1.6.2)




\spg{Betti setting}

\begin{stp}
	$k=\bC$
\end{stp}

Betti geometric Langlands: the following functor is an equivalence.
\[
	\begin{tikzcd}
		\Shv_{\frac{1}{2}, \text{Nilp}}^{\text{Betti}}(\Bun_{G})\ar[r, "\bL_{G}^{\text{Betti}}"] & \IndCoh_{\text{Nilp}}(\mathrm{LS}_{\oH{G}}^{\text{Betti}})
	\end{tikzcd}
\]

The implication is

Full de Rham = Restricted de Rham = Restricted Betti = Full Betti

middle arrow is Riemann-Hilbert

Proof uses Kac-Moody localization


\[
	\Bu: \text{D-Mod}\alrs \text{D-Mod}:\Bu^{R}
\]
