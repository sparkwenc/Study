\begin{qst}
  Let $X$ be a scheme, $X(R)$ carries a Zariski topology naturally, but what if $R$ is a topological ring, especically if $R=\bA_{F}$ for some global field $F$?
\end{qst}

\section{Affine case}

\begin{prp}\label{prp:AffineCase}
  Let $R$ be a topological ring, and any affine finite type $R$ scheme $X=\Spec A$, $X(R)$ is endowed with the weakest topology such that $X(R)\ar R$ is continuous for any $f\in A$, and this is the unique topology satisfying
  \begin{enumerate}
    \item functorial in $X$,
    \item compatible with fiber product,
    \item carries closed immersion into topological embeddings,
    \item $\bA_{R}^{1}(R)=R$ as topological space.
  \end{enumerate}
  Moreover,
  \begin{enumerate}
    \item if $R$ is Hausdorff, $X(R)$ is Hausdorff,
    \item if $R$ is locally compact, $X(R)$ is locally compact.
  \end{enumerate}
\end{prp}

\begin{prf}
  Uniqueness: direct from the properties, $X\ahr\Spec R[t_{1},\ldots,t_{n}]$ induces closed embedding $X(R)\ahr R^{n}$. Now $X(R)$ carries the subspace topology, and is Hausdorff/locally compact if $R$ is.


  Existence: it suffices to see $X(R)=\Hom_{R}(A,R)$ is given the subspace topology on $\Hom(A,R)$ with product topology.


  To show the properties are satisfied, it suffices to show 3, and $h:A'\atr A$ induces topological embedding $R^{A}\ar R^{A'}$ via $(r_{a})\mapsto (r_{h(a')})$.
\end{prf}

\begin{exg}
  Continuous $R\ar R'$ induces continuous $X(R)\ar X(R')$,
  \begin{enumerate}
    \item $F\ahr\bA_{F}$ discrete, so $X(F)\ahr X(\bA_{F})$ discrete,
    \item $R\ar R'$ is an finite extension, $X'$ affine $R'$-scheme of finite type, $X=\Res_{R'/R}(X')$, then $X'(R')=X(R)$ is a topological isomorphism.
  \end{enumerate}
\end{exg}


\section{General case}

\begin{qst}
  Note the construction in Proposition~\ref{prp:AffineCase} does not necessarily give open embedding $U(R)\ar X(R)$ if $U\ar R$ is open affine. But can generalize this if $R\xT$ is open in $R$.
\end{qst}

\begin{prp}\label{prp:GeneralCase}
  Let $R$ be a local topological ring with $R\xT$ open and has continuous inverse, $X$ be a $R$-scheme locally of finite type. Then there is a unique way to topologize $X(R)$ satisfying
  \begin{enumerate}
    \item functorial in $X$,
    \item carries closed/open immersion into closed/open embeddings,
    \item compatible with fiber products,
    \item $\bA_{R}^{1}(R)=R$ as topological ring.
  \end{enumerate}
  Moreover,
  \begin{enumerate}
    \item agrees with construction in Proposition~\ref{prp:AffineCase}.
    \item $X(R)$ is Hausdorff if $X$ is separated over $R$,
    \item $X(R)$ is locally compact if $R$ is locally compact and Hausdorff.
  \end{enumerate}
\end{prp}

\begin{prf}
  It suffices to show any open immersion between affine $R$-schemes of finite type $U\ar X$ induces open embedding $U(R)\ar X(R)$. To do this, first show this for principal open $U=\Spec A_{f}\ar X=\Spec A$, then show if $U_{i}$ cover $X$, $U_{i}(R)$ cover $X(R)$.

  And conclude with gluing argument.


  For 1, note $U=\bG_{m,R}\times_{\bA_{R}^{1}}X$, it suffices to do for $U=\bG_{m,R}, X=\bA_{R}^{1}$, this follows from hypothesis $R\xT$ being open and inversion being continuous.

  For 2, note $X(R)=\bigcup_{i}U_{i}(R)$ comes from $R$ being local: any open $U_{i}(R)$ containing closed point of $\Spec R$ also contains $\Spec R$.
\end{prf}

\begin{exg}
  Let $R$ be a local field or a compact discrete valuation ring, $X$ a $R$-scheme locally of finite type. Then $X(R)$ is locally compact and Hausdorff.
\end{exg}



\section{Topology on adelic points}
\begin{qst}
  Let $F$ be a global field, $R=\bA_{F}$, how to topologize $X(R)$?
\end{qst}
\begin{thm}
  If $\{A_{i}\}$ is a directed systen of rings, $A=\varinjlim A_{i}$, $X$ a finitely presented $A$-scheme.
  \begin{enumerate}
    \item $\exists i_{0}$ and a finitely presented $A_{i_{0}}$-scheme $X_{i_{0}}$, such that $X_{i_{0}}\times_{A_{i_{0}}}A=X$, moreover the following map of set is bijective
          \[
            \varinjlim\Hom_{A_{i}}(X_{i},Y_{i})\ar\Hom_{A}(X,Y).
          \]
    \item A map $f_{i_{0}}:X_{i_{0}}\ar Y_{i_{0}}$ has property $P$ after base change to some $A_{i}, i\geq i_{0}$ iff the induced morphism $f:X\ar Y$ has property $P$.
          \[
            P=\bmat{\text{closed immersion, open immersion, separated, proper, smooth, affine, flat, finite,}\\\text{geometrically connected of pure dimension $d$, non-empty fibers}}.
          \]
    \item If finitely presented $A_{i_{0}}$-scheme $X_{i_{0}}, X_{i_{0}}'$ both become $X$ after base change to $A$, then there exists $i\geq i_{0}$ such that $X_{i}\cong X_{i}'$, and such isomorphism is identified after base change to some $i'\geq i$.
  \end{enumerate}
\end{thm}


\begin{exg}
  Let $F$ be a global field, $X$ a finite type $F$-scheme, pick a finite set $S$ such that there is a model $X_{S}\ar\Spec\cO_{F,S}$ with generic fiber $X$, then
  \[
    \varinjlim_{S'\supset S}X_{S'}(\bA_{F,S'})=\varinjlim_{S'\supset S}X_{S}(\bA_{F,S'})\cong X_{S}(\bA_{F})=X(\bA_{F}).
  \]

  If $X_{S}$ is a finite type $\cO_{F,S}$-scheme, then the following is a homeomorphism
  \[
    X_{S}(\bA_{F,S})\ar\prod_{v\in S}X_{v}(F_{v})\times\prod_{v\notin S}X_{S,v}(\cO_{v}).
  \]
\end{exg}

\begin{cor}
  If $X_{S}$ is a finite type $\cO_{S}$-scheme, then the topological space $X_{S}(\bA_{F,S})$ is locally compact and is Hausdorff if $X_{S}$ is separated.


  If $F\ar F'$ is a finite extension of global fields, $X'$ a quasi-projective $F'$-scheme, then $X=\Res_{F'/F}X'$ is separated and finite type over $F$. Now $X(\bA_{F})=X'(\bA_{F'})$ is an isomorphism.
\end{cor}


\begin{prp}
  Let $X\ar Y$ be a proper morphism between separated $F$-schemes of finite type, then the induced $X(\bA_{F})\ar Y(\bA_{F})$ between locally compact Hausdorff spaces is topologically proper. In particular if $X$ is proper over $F$ then $X(\bA_{F})$ is compact.
\end{prp}


\begin{thm}
  Let $X\ar Y$ be a smooth surjective morphism between separated $F$-schemes of finite type with geometrically connected fibers, then the induced $X(\bA_{F})\ar Y(\bA_{F})$ is open.
\end{thm}
