\ssc{Self-dual sparsity}
Ref:~\cite{Rohrlich2013}.

\begin{enr}
	\item Show that Fontaine-Mazur conjectures imply that essentially self-dual motives $M$ of weight $w$ has density zero among all motives. For the reading list
	\begin{enr}
		\item \cite{FM1995,Emerton2006,Kisin2009FontaineMazur}
	\end{enr}
	\item Show that the symplectic and orthogonal Artin representations of a fixed number field $F$ of dimension $n$ have density $0$, among all Artin representations of dimension $n$. For the reading list
	\begin{enr}
		\item \cite{Rohrlich2013,Karnataki2017}
		\item \cite{Malle2002,Malle2004,Kluners2005,Turkelli2015}
		\item \cite{ASVW2021,CDO2002,Zagier1997}
	\end{enr}
	\item Show that the automorphic forms on $\GL_{n}/\bQ$ lifted from symplectic and orthogonal groups have density $0$, among all automorphic forms on $\GL_{n}/\bQ$. One step back, prove for Maass forms. For the reading list
	\begin{enr}
		\item \cite{Iwaniec1997,Bump1997}
		\item \cite{Kala2014,BLS2020}
	\end{enr}
	For studies of the asymptotics of the dimension of the space of Maass forms,
	\begin{enr}
		\item \cite{LV2007,Muller2007,FLM2015,MT2021}
	\end{enr}
\end{enr}
